@ Load the twisting corridor environment
LOAD 1

@ Main loop to keep the robot moving
WHILE TRUE
    @ Escape immediately if at exit
    IF AT_EXIT
        OPEN_DOOR
    END

    @ If we find the key, pick it up
    IF ON_KEY
        PICK_KEY
    END

    @ If we are at the door, try to open (will succeed only with key)
    IF AT_DOOR
        OPEN_DOOR
    END

    @ Left-hand wall following
    TURN_LEFT
    IF FRONT_IS_CLEAR
        MOVE_FORWARD
    OTHERWISE
        TURN_RIGHT
        IF FRONT_IS_CLEAR
            MOVE_FORWARD
        OTHERWISE
            TURN_RIGHT
        END
    END
END

END
